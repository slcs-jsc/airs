112

115

119

121
122 atm_t *atm;
123
124 obs_t *obs;
125
126 pert_t *pert;
127
128 wave_t *wave;
129
130 model_t *model;
131
132 /* ------------------------------------------------------------
133 Get control parameters...
134 ------------------------------------------------------------ */
135
136 /* Check arguments... */
137 if (argc < 8)
138 ERRMSG("Give parameters: <ctl> <model> <model.nc> <pert.nc>"
139 " <wave_airs.tab> <wave_model.tab> <wave_airs.nc> <wave_model.nc>");
140
141 /* Read control parameters... */
142 read_ctl(argc, argv, &ctl);
143 scan_ctl(argc, argv, "PERTNAME", -1, "4mu", pertname);
144 scan_ctl(argc, argv, "KERNEL", -1, "-", kernel);
145 const int extpol = (int) scan_ctl(argc, argv, "EXTPOL", -1, "0", NULL);
146 const int slant = (int) scan_ctl(argc, argv, "SLANT", -1, "1", NULL);
147 const double t_ovp = scan_ctl(argc, argv, "T_OVP", -1, "", NULL);
148
149 /* Set control parameters... */
150 ctl.write_bbt = 1;
151
152 /* Initialize look-up tables... */
153 tbl_t *tbl = read_tbl(&ctl);
154
155 /* ------------------------------------------------------------
156 Read model data...
157 ------------------------------------------------------------ */
158
159 /* Allocate... */
160 ALLOC(model, model_t, 1);
161 ALLOC(help, float,
162 NLON * NLAT * NZ);
163
164 /* Open file... */
165 printf("Read %s data: %s\n", argv[2], argv[3]);
166 NC(nc_open(argv[3], NC_NOWRITE, &ncid));
167
168 /* Check time... */
169 LOG(2, "Check time...");
170 if (nc_inq_dimid(ncid, "time", &dimid) != NC_NOERR)
171 NC(nc_inq_dimid(ncid, "time", &dimid));
172 NC(nc_inq_dimlen(ncid, dimid, &rs));
173 if (rs != 1)
174 ERRMSG("Only one time step is allowed!");
175
176 /* Read latitudes... */
177 LOG(2, "Read latitudes...");
178 if (nc_inq_dimid(ncid, "lat", &dimid) != NC_NOERR)
179 NC(nc_inq_dimid(ncid, "latitude", &dimid));
180 NC(nc_inq_dimlen(ncid, dimid, &rs));
181 model->nlat = (int) rs;
182 if (model->nlat > NLAT)
183 ERRMSG("Too many latitudes!");
184 if (nc_inq_varid(ncid, "lat", &varid) != NC_NOERR)
185 NC(nc_inq_varid(ncid, "latitude", &varid));
186 NC(nc_get_var_double(ncid, varid, model->lat));
187
188 /* Read longitudes... */
189 LOG(2, "Read longitudes...");
190 if (nc_inq_dimid(ncid, "lon", &dimid) != NC_NOERR)
191 NC(nc_inq_dimid(ncid, "longitude", &dimid));
192 NC(nc_inq_dimlen(ncid, dimid, &rs));
193 model->nlon = (int) rs;
194 if (model->nlon > NLON)
195 ERRMSG("Too many longitudes!");
196 if (nc_inq_varid(ncid, "lon", &varid))
197 NC(nc_inq_varid(ncid, "longitude", &varid));
198 NC(nc_get_var_double(ncid, varid, model->lon));
199
200 /* Read ICON data... */
201 if (strcasecmp(argv[2], "icon") == 0) {
202
203 /* Get height levels... */
204 LOG(2, "Read levels...");
205 NC(nc_inq_dimid(ncid, "height", &dimid));
206 NC(nc_inq_dimlen(ncid, dimid, &rs));
207 model->nz = (int) rs;
208 if (model->nz > NZ)
209 ERRMSG("Too many altitudes!");
210
211 /* Read height... */
212 LOG(2, "Read height...");
213 NC(nc_inq_varid(ncid, "z_mc", &varid));
214 NC(nc_get_var_float(ncid, varid, help));
215 for (int ilon = 0; ilon < model->nlon; ilon++)
216 for (int ilat = 0; ilat < model->nlat; ilat++)
217 for (int iz = 0; iz < model->nz; iz++)
218 model->z[ilon][ilat][iz] =
219 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
220 1e3);
221
222 /* Read temperature... */
223 LOG(2, "Read temperature...");
224 NC(nc_inq_varid(ncid, "temp", &varid));
225 NC(nc_get_var_float(ncid, varid, help));
226 for (int ilon = 0; ilon < model->nlon; ilon++)
227 for (int ilat = 0; ilat < model->nlat; ilat++)
228 for (int iz = 0; iz < model->nz; iz++)
229 model->t[ilon][ilat][iz] =
230 help[(iz * model->nlat + ilat) * model->nlon + ilon];
231
232 /* Read pressure... */
233 LOG(2, "Read pressure...");
234 NC(nc_inq_varid(ncid, "pres", &varid));
235 NC(nc_get_var_float(ncid, varid, help));
236 for (int ilon = 0; ilon < model->nlon; ilon++)
237 for (int ilat = 0; ilat < model->nlat; ilat++)
238 for (int iz = 0; iz < model->nz; iz++)
239 model->p[ilon][ilat][iz] =
240 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
241 1e2);
242 }
243
244 /* Read IFS data... */
245 else if (strcasecmp(argv[2], "ifs") == 0) {
246
247 /* Get height levels... */
248 LOG(2, "Read levels...");
249 NC(nc_inq_dimid(ncid, "lev_2", &dimid));
250 NC(nc_inq_dimlen(ncid, dimid, &rs));
251 model->nz = (int) rs;
252 if (model->nz > NZ)
253 ERRMSG("Too many altitudes!");
254
255 /* Read height... */
256 LOG(2, "Read height...");
257 NC(nc_inq_varid(ncid, "gh", &varid));
258 NC(nc_get_var_float(ncid, varid, help));
259 for (int ilon = 0; ilon < model->nlon; ilon++)
260 for (int ilat = 0; ilat < model->nlat; ilat++)
261 for (int iz = 0; iz < model->nz; iz++)
262 model->z[ilon][ilat][iz] =
263 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
264 1e3);
265
266 /* Read temperature... */
267 LOG(2, "Read temperature...");
268 NC(nc_inq_varid(ncid, "t", &varid));
269 NC(nc_get_var_float(ncid, varid, help));
270 for (int ilon = 0; ilon < model->nlon; ilon++)
271 for (int ilat = 0; ilat < model->nlat; ilat++)
272 for (int iz = 0; iz < model->nz; iz++)
273 model->t[ilon][ilat][iz] =
274 help[(iz * model->nlat + ilat) * model->nlon + ilon];
275
276 /* Read surface pressure... */
277 LOG(2, "Read surface pressure...");
278 NC(nc_inq_varid(ncid, "lnsp", &varid));
279 NC(nc_get_var_float(ncid, varid, help));
280 for (int ilon = 0; ilon < model->nlon; ilon++)
281 for (int ilat = 0; ilat < model->nlat; ilat++)
282 model->ps[ilon][ilat] = (float) exp(help[ilat * model->nlon + ilon]);
283
284 /* Read grid coefficients... */
285 LOG(2, "Read grid coefficients...");
286 NC(nc_inq_varid(ncid, "hyam", &varid));
287 NC(nc_get_var_double(ncid, varid, hyam));
288 NC(nc_inq_varid(ncid, "hybm", &varid));
289 NC(nc_get_var_double(ncid, varid, hybm));
290
291 /* Calculate pressure... */
292 LOG(2, "Calculate pressure...");
293 for (int ilon = 0; ilon < model->nlon; ilon++)
294 for (int ilat = 0; ilat < model->nlat; ilat++)
295 for (int iz = 0; iz < model->nz; iz++)
296 model->p[ilon][ilat][iz]
297 = (float) ((hyam[iz] + hybm[iz] * model->ps[ilon][ilat]) / 100.);
298 }
299
300 /* Read UM data... */
301 else if (strcasecmp(argv[2], "um") == 0) {
302
303 /* Get height levels... */
304 LOG(2, "Read levels...");
305 if (nc_inq_dimid(ncid, "RHO_TOP_eta_rho", &dimid) != NC_NOERR)
306 NC(nc_inq_dimid(ncid, "RHO_eta_rho", &dimid));
307 NC(nc_inq_dimlen(ncid, dimid, &rs));
308 model->nz = (int) rs;
309 if (model->nz > NZ)
310 ERRMSG("Too many altitudes!");
311
312 /* Read height... */
313 LOG(2, "Read height...");
314 if (nc_inq_varid(ncid, "STASH_m01s15i102_2", &varid) != NC_NOERR)
315 NC(nc_inq_varid(ncid, "STASH_m01s15i102", &varid));
316 NC(nc_get_var_float(ncid, varid, help));
317 for (int ilon = 0; ilon < model->nlon; ilon++)
318 for (int ilat = 0; ilat < model->nlat; ilat++)
319 for (int iz = 0; iz < model->nz; iz++)
320 model->z[ilon][ilat][iz] =
321 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
322 1e3);
323
324 /* Read temperature... */
325 LOG(2, "Read temperature...");
326 NC(nc_inq_varid(ncid, "STASH_m01s30i004", &varid));
327 NC(nc_get_var_float(ncid, varid, help));
328 for (int ilon = 0; ilon < model->nlon; ilon++)
329 for (int ilat = 0; ilat < model->nlat; ilat++)
330 for (int iz = 0; iz < model->nz; iz++)
331 model->t[ilon][ilat][iz] =
332 help[(iz * model->nlat + ilat) * model->nlon + ilon];
333
334 /* Read pressure... */
335 LOG(2, "Read pressure...");
336 NC(nc_inq_varid(ncid, "STASH_m01s00i407", &varid));
337 NC(nc_get_var_float(ncid, varid, help));
338 for (int ilon = 0; ilon < model->nlon; ilon++)
339 for (int ilat = 0; ilat < model->nlat; ilat++)
340 for (int iz = 0; iz < model->nz; iz++)
341 model->p[ilon][ilat][iz] =
342 0.01f * help[(iz * model->nlat + ilat) * model->nlon + ilon];
343 }
344
345 /* Read WRF data... */
346 else if (strcasecmp(argv[2], "wrf") == 0) {
347
348 /* Get height levels... */
349 LOG(2, "Read levels...");
350 NC(nc_inq_dimid(ncid, "bottom_top", &dimid));
351 NC(nc_inq_dimlen(ncid, dimid, &rs));
352 model->nz = (int) rs;
353 if (model->nz > NZ)
354 ERRMSG("Too many altitudes!");
355
356 /* Read height... */
357 LOG(2, "Read height...");
358 NC(nc_inq_varid(ncid, "z", &varid));
359 NC(nc_get_var_float(ncid, varid, help));
360 for (int ilon = 0; ilon < model->nlon; ilon++)
361 for (int ilat = 0; ilat < model->nlat; ilat++)
362 for (int iz = 0; iz < model->nz; iz++)
363 model->z[ilon][ilat][iz] =
364 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
365 1e3);
366
367 /* Read temperature... */
368 LOG(2, "Read temperature...");
369 NC(nc_inq_varid(ncid, "tk", &varid));
370 NC(nc_get_var_float(ncid, varid, help));
371 for (int ilon = 0; ilon < model->nlon; ilon++)
372 for (int ilat = 0; ilat < model->nlat; ilat++)
373 for (int iz = 0; iz < model->nz; iz++)
374 model->t[ilon][ilat][iz] =
375 help[(iz * model->nlat + ilat) * model->nlon + ilon];
376
377 /* Read pressure... */
378 LOG(2, "Read pressure...");
379 NC(nc_inq_varid(ncid, "p", &varid));
380 NC(nc_get_var_float(ncid, varid, help));
381 for (int ilon = 0; ilon < model->nlon; ilon++)
382 for (int ilat = 0; ilat < model->nlat; ilat++)
383 for (int iz = 0; iz < model->nz; iz++)
384 model->p[ilon][ilat][iz] =
385 (float) (help[(iz * model->nlat + ilat) * model->nlon + ilon] /
386 1e2);
387 }
388
389 else
390 ERRMSG("Model type not supported!");
391
392 /* Close file... */
393 NC(nc_close(ncid));
394
395 /* Free... */
396 free(help);
397
398 /* Check data... */
399 LOG(2, "Checking...");
400 for (int ilon = 0; ilon < model->nlon; ilon++)
401 for (int ilat = 0; ilat < model->nlat; ilat++)
402 for (int iz = 0; iz < model->nz; iz++)
403 if (model->t[ilon][ilat][iz] <= 100
404 || model->t[ilon][ilat][iz] >= 400) {
405 model->p[ilon][ilat][iz] = GSL_NAN;
406 model->t[ilon][ilat][iz] = GSL_NAN;
407 model->z[ilon][ilat][iz] = GSL_NAN;
408 }
409
410 /* Smoothing of model data... */
411 LOG(2, "Smoothing...");
412 smooth(model);
413
414 /* Calcuate height above the surface... */
415 for (int ilon = 0; ilon < model->nlon; ilon++)
416 for (int ilat = 0; ilat < model->nlat; ilat++) {
417 float z0 = model->z[ilon][ilat][0];
418 for (int iz = 1; iz < model->nz; iz++)
419 z0 = GSL_MIN(z0, model->z[ilon][ilat][iz]);
420 for (int iz = 0; iz < model->nz; iz++)
421 model->z[ilon][ilat][iz] -= z0;
422 }
423
424 /* Write info... */
425 for (int iz = 0; iz < model->nz; iz++)
426 printf("section_height: %d %g %g %g %g %g\n", iz,
427 model->z[model->nlon / 2][model->nlat / 2][iz],
428 model->lon[model->nlon / 2], model->lat[model->nlat / 2],
429 model->p[model->nlon / 2][model->nlat / 2][iz],
430 model->t[model->nlon / 2][model->nlat / 2][iz]);
431 for (int ilon = 0; ilon < model->nlon; ilon++)
432 printf("section_west_east: %d %g %g %g %g %g\n", ilon,
433 model->z[ilon][model->nlat / 2][model->nz / 2],
434 model->lon[ilon], model->lat[model->nlat / 2],
435 model->p[ilon][model->nlat / 2][model->nz / 2],
436 model->t[ilon][model->nlat / 2][model->nz / 2]);
437 for (int ilat = 0; ilat < model->nlat; ilat++)
438 printf("section_north_south: %d %g %g %g %g %g\n", ilat,
439 model->z[model->nlon / 2][ilat][model->nz / 2],
440 model->lon[model->nlon / 2], model->lat[ilat],
441 model->p[model->nlon / 2][ilat][model->nz / 2],
442 model->t[model->nlon / 2][ilat][model->nz / 2]);
443
444 /* ------------------------------------------------------------
445 Read AIRS perturbation data...
446 ------------------------------------------------------------ */
447
448 /* Allocate... */
449 ALLOC(atm, atm_t, 1);
450 ALLOC(obs, obs_t, 1);
451 ALLOC(pert, pert_t, 1);
452 ALLOC(wave, wave_t, 1);
453
454 /* Read perturbation data... */
455 read_pert(argv[4], pertname, pert);
456
457 /* Find track range... */
458 for (int itrack = 0; itrack < pert->ntrack; itrack++) {
459 if (pert->time[itrack][44] < t_ovp - 720 || itrack == 0)
460 track0 = itrack;
461 track1 = itrack;
462 if (pert->time[itrack][44] > t_ovp + 720)
463 break;
464 }
465
466 /* Convert to wave analysis struct... */
467 pert2wave(pert, wave, track0, track1, 0, pert->nxtrack - 1);
468
469 /* Estimate background... */
470 background_poly(wave, 5, 0);
471
472 /* Compute variance... */
473 variance(wave, var_dh);
474
475 /* Write observation wave struct... */
476 write_wave(argv[5], wave);
477 write_nc(argv[7], wave);
478
479 /* ------------------------------------------------------------
480 Run forward model...
481 ------------------------------------------------------------ */
482
483 /* Loop over AIRS geolocations... */
484 for (int itrack = track0; itrack <= track1; itrack++)
485 for (int ixtrack = 0; ixtrack < pert->nxtrack; ixtrack++) {
486
487 /* Write info... */
488 if (ixtrack == 0)
489 printf("Compute track %d / %d ...\n", itrack - track0 + 1,
490 track1 - track0 + 1);
491
492 /* Set observation data... */
493 obs->nr = 1;
494 obs->obsz[0] = 705;
495 obs->obslon[0] = pert->lon[itrack][44];
496 obs->obslat[0] = pert->lat[itrack][44];
497 obs->vpz[0] = 0;
498 obs->vplon[0] = pert->lon[itrack][ixtrack];
499 obs->vplat[0] = pert->lat[itrack][ixtrack];
500
501 /* Get Cartesian coordinates... */
502 geo2cart(obs->obsz[0], obs->obslon[0], obs->obslat[0], xo);
503 geo2cart(obs->vpz[0], obs->vplon[0], obs->vplat[0], xs);
504
505 /* Set profile for atmospheric data... */
506 if (slant) {
507 atm->np = 0;
508 for (double f = 0.0; f <= 1.0; f += 0.0002) {
509 xm[0] = f * xo[0] + (1 - f) * xs[0];
510 xm[1] = f * xo[1] + (1 - f) * xs[1];
511 xm[2] = f * xo[2] + (1 - f) * xs[2];
512 cart2geo(xm, &atm->z[atm->np], &atm->lon[atm->np],
513 &atm->lat[atm->np]);
514 atm->time[atm->np] = pert->time[itrack][ixtrack];
515 if (atm->z[atm->np] < 10)
516 continue;
517 else if (atm->z[atm->np] > 90)
518 break;
519 else if ((++atm->np) >= NP)
520 ERRMSG("Too many ray path data points!");
521 }
522 } else {
523 atm->np = 0;
524 for (double f = 10.0; f <= 90.0; f += 0.2) {
525 atm->time[atm->np] = pert->time[itrack][ixtrack];
526 atm->z[atm->np] = f;
527 atm->lon[atm->np] = pert->lon[itrack][ixtrack];
528 atm->lat[atm->np] = pert->lat[itrack][ixtrack];
529 if ((++atm->np) >= NP)
530 ERRMSG("Too many ray path data points!");
531 }
532 }
533
534 /* Initialize with climatological data... */
535 climatology(&ctl, atm);
536
537 /* Interpolate model data... */
538 for (int ip = 0; ip < atm->np; ip++)
539 intpol(model, atm->z[ip], atm->lon[ip], atm->lat[ip],
540 &atm->p[ip], &atm->t[ip]);
541
542 /* Check profile... */
543 int okay = 1;
544 for (int ip = 0; ip < atm->np; ip++)
545 if (!gsl_finite(atm->p[ip]) || !gsl_finite(atm->t[ip]))
546 okay = 0;
547 if (!okay)
548 pert->bt[itrack][ixtrack] = GSL_NAN;
549 else {
550
551 /* Use kernel function... */
552 if (kernel[0] != '-') {
553
554 /* Read kernel function... */
555 if (!init) {
556 init = 1;
557 read_shape(kernel, kz, kw, &nk);
558 if (kz[0] > kz[1])
559 ERRMSG("Kernel function must be ascending!");
560 }
561
562 /* Calculate mean temperature... */
563 pert->bt[itrack][ixtrack] = wsum = 0;
564 for (int ip = 0; ip < atm->np; ip++)
565 if (atm->z[ip] >= kz[0] && atm->z[ip] <= kz[nk - 1]) {
566 const int iz = locate_irr(kz, nk, atm->z[ip]);
567 const double w =
568 LIN(kz[iz], kw[iz], kz[iz + 1], kw[iz + 1], atm->z[ip]);
569 pert->bt[itrack][ixtrack] += w * atm->t[ip];
570 wsum += w;
571 }
572 pert->bt[itrack][ixtrack] /= wsum;
573 }
574
575 /* Use radiative transfer model... */
576 else {
577
578 /* Run forward model... */
579 formod(&ctl, tbl, atm, obs);
580
581 /* Get mean brightness temperature... */
582 pert->bt[itrack][ixtrack] = 0;
583 for (int id = 0; id < ctl.nd; id++)
584 pert->bt[itrack][ixtrack] += obs->rad[id][0] / ctl.nd;
585 }
586 }
587 }
588
589 /* Extrapolate... */
590 if (extpol)
591 for (int itrack = track0; itrack <= track1; itrack++) {
592 for (int ixtrack = 1; ixtrack < pert->nxtrack; ixtrack++)
593 if (!gsl_finite(pert->bt[itrack][ixtrack]))
594 pert->bt[itrack][ixtrack] = pert->bt[itrack][ixtrack - 1];
595 for (int ixtrack = pert->nxtrack - 2; ixtrack >= 0; ixtrack--)
596 if (!gsl_finite(pert->bt[itrack][ixtrack]))
597 pert->bt[itrack][ixtrack] = pert->bt[itrack][ixtrack + 1];
598 }
599
600 /* ------------------------------------------------------------
601 Write model perturbations...
602 ------------------------------------------------------------ */
603
604 /* Convert to wave analysis struct... */
605 pert2wave(pert, wave, track0, track1, 0, pert->nxtrack - 1);
606
607 /* Estimate background... */
608 background_poly(wave, 5, 0);
609
610 /* Compute variance... */
611 variance(wave, var_dh);
612
613 /* Write observation wave struct... */
614 write_wave(argv[6], wave);
615 write_nc(argv[8], wave);
616
617 /* Free... */
618 free(atm);
619 free(obs);
620 free(pert);
621 free(wave);
622 free(tbl);
623
624 return EXIT_SUCCESS;
625}

627/************************************************************************/
628
629void intpol(
630 model_t *model,
631 double z,
632 double lon,
633 double lat,
634 double *p,
635 double *t) {
636
637 double zd[NZ];
638
639 int iz;
640
641 /* Adjust longitude... */
642 if (model->lon[model->nlon - 1] > 180)
643 if (lon < 0)
644 lon += 360;
645
646 /* Check horizontal range... */
647 if (lon < model->lon[0]
648 || lon > model->lon[model->nlon - 1]
649 || lat < GSL_MIN(model->lat[0], model->lat[model->nlat - 1])

750void smooth(
751 model_t *model) {
752
753 static float hp[NLON][NLAT], ht[NLON][NLAT], hz[NLON][NLAT];
754
755 static double wx[10], wy[10];
756
757 const int dlon = 3, dlat = 3;
758
759 /* Set weights... */
760 const double dy = RE * M_PI / 180. * fabs(model->lat[1] - model->lat[0]);
761 for (int ilat = 0; ilat <= dlat; ilat++)
762 wy[ilat] = exp(-0.5 * POW2(ilat * dy * 2.35482 / 20.));
763
764 /* Loop over height levels... */
765 for (int iz = 0; iz < model->nz; iz++) {
766
767 /* Write info... */
768 printf("Smoothing level %d / %d ...\n", iz + 1, model->nz);
769
770 /* Copy data... */
771 for (int ilon = 0; ilon < model->nlon; ilon++)
772 for (int ilat = 0; ilat < model->nlat; ilat++) {
773 hp[ilon][ilat] = model->p[ilon][ilat][iz];
774 ht[ilon][ilat] = model->t[ilon][ilat][iz];
775 hz[ilon][ilat] = model->z[ilon][ilat][iz];
776 }
777
778 /* Loop over latitudes... */
779 for (int ilat = 0; ilat < model->nlat; ilat++) {
780
781 /* Set weights... */
782 const double dx =
783 RE * M_PI / 180. * cos(model->lat[ilat] * M_PI / 180.) *
784 fabs(model->lon[1] - model->lon[0]);
785 for (int ilon = 0; ilon <= dlon; ilon++)
786 wx[ilon] = exp(-0.5 * POW2(ilon * dx * 2.35482 / 20.));
787
788 /* Loop over longitudes... */
789 for (int ilon = 0; ilon < model->nlon; ilon++) {
790 float wsum = 0;
791 model->p[ilon][ilat][iz] = 0;
792 model->t[ilon][ilat][iz] = 0;
793 model->z[ilon][ilat][iz] = 0;
794 for (int ilon2 = GSL_MAX(ilon - dlon, 0);
795 ilon2 <= GSL_MIN(ilon + dlon, model->nlon - 1); ilon2++)
796 for (int ilat2 = GSL_MAX(ilat - dlat, 0);
797 ilat2 <= GSL_MIN(ilat + dlat, model->nlat - 1); ilat2++) {
798 float w = (float) (wx[abs(ilon2 - ilon)] * wy[abs(ilat2 - ilat)]);
799 model->p[ilon][ilat][iz] += w * hp[ilon2][ilat2];
800 model->t[ilon][ilat][iz] += w * ht[ilon2][ilat2];
801 model->z[ilon][ilat][iz] += w * hz[ilon2][ilat2];
802 wsum += w;
803 }
804 model->p[ilon][ilat][iz] /= wsum;
805 model->t[ilon][ilat][iz] /= wsum;
806 model->z[ilon][ilat][iz] /= wsum;
807 }
808 }
809 }
810}
