491 itrack2++)
492 for (int ixtrack2 = GSL_MAX(ixtrack - iradius, 0);
493 ixtrack2 <= GSL_MIN(ixtrack + iradius, pert->nxtrack - 1);

515 min[ix][iy] = GSL_MIN(min[ix][iy], pert->pt[itrack][ixtrack]);

527 bt_8mu_max[ix][iy] = pert->dc[itrack][ixtrack];
